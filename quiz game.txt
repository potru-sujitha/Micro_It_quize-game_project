  <script>
    const quizData = [
      {
        question: "What is the capital of France?",
        options: ["Paris", "London", "Rome", "Berlin"],
        answer: "Paris"
      },
      {
        question: "What is 2 + 2?",
        options: ["3", "4", "5", "2"],
        answer: "4"
      },
      {
        question: "Which planet is known as the Red Planet?",
        options: ["Earth", "Saturn", "Mars", "Jupiter"],
        answer: "Mars"
      },
      {
        question: "Who wrote 'Harry Potter'?",
        options: ["J.R.R. Tolkien", "William Shakespeare", "J.K. Rowling", "Mark Twain"],
        answer: "J.K. Rowling"
      }
    ];

    let currentQuestion = 0;

    function loadQuestion() {
      const q = quizData[currentQuestion];
      document.getElementById('question').innerText = q.question;
      const optionsDiv = document.getElementById('options');
      optionsDiv.innerHTML = '';

      q.options.forEach(option => {
        const button = document.createElement('button');
        button.innerText = option;
        button.onclick = () => checkAnswer(option);
        optionsDiv.appendChild(button);
      });

      document.getElementById('result').innerText = '';
    }

    function checkAnswer(selected) {
      const correct = quizData[currentQuestion].answer;
      const resultDiv = document.getElementById('result');
      if (selected === correct) {
        resultDiv.innerText = "‚úÖ Correct!";
        resultDiv.style.color = "lightgreen";
      } else {
        resultDiv.innerText = `‚ùå Wrong! Correct answer is: ${correct}`;
        resultDiv.style.color = "salmon";
      }
    }

    function nextQuestion() {
      currentQuestion++;
      if (currentQuestion < quizData.length) {
        loadQuestion();
      } else {
        document.getElementById('quiz').innerHTML = `<h2>üéä Quiz Completed! Great job! üéä</h2>`;
      }
    }

    // Initialize
    loadQuestion();
  </script>
